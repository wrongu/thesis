                358         function fullPathName = getFullPathName(fileName)
                359         % Given a fileName, relative file path, or full file path, return
                360         % the Full path (i.e. fullpath/fileName.ext) to the given file
                361         
                362             % First check the MATLAB path for the file.
                363             whichFileName = which(fileName);
                364             if ~strcmp(whichFileName, '')
                365                 fullPathName = whichFileName;
                366                 return;
                367             end
                368 
                369             % We have file not on the MATLAB path,
                370             % get the full path using fileattrib.
                371             [stat info] = fileattrib(fileName);
                372             if ~stat
                373                 error('MATLAB:VideoReader:fileNotFound', ...
                374                       VideoReader.getError('MATLAB:VideoReader:fileNotFound'));
                375             end
                376 
                377             fullPathName = info.Name;
                378             if strcmp(fullPathName,'')
                379                 error('MATLAB:VideoReader:fileNotFound', ...
                380                       VideoReader.getError('MATLAB:VideoReader:fileNotFound'));
                381             end
                382         end
                383     end
